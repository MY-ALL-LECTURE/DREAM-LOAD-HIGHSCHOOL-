const unsigned int led[5] = {2,3,4,5,6};

void setup() {
  for(int x=0 ; x<5 ; x++){
    pinMode(led[x],OUTPUT);
  }
}
void loop() {
  for(int i=0;i<5;i++){  
      //ALL OFF
      for(int x=0;x<=5;x++){
        digitalWrite(led[x],LOW);
      }  
      digitalWrite(led[i], HIGH);
      delay(500);
  }
}
