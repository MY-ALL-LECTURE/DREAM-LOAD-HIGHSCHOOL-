const int ledPin = 13;
void setup() {
  Serial.begin(9600);
  pinMode(ledPin,OUTPUT);
}

void loop() {
  if(Serial.available()){
   	char inputVal = Serial.read();
    Serial.println(inputVal);
    if(inputVal == '1'){
      digitalWrite(ledPin,HIGH);
      Serial.println("LED:ON");
    }
    else if(inputVal == '0'){
      digitalWrite(ledPin,LOW);
      Serial.println("LED:OFF");
    }
  }
  delay(100);


}
